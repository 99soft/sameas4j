22  package org.nnsoft.sameas4j;
23  
24  import java.net.URI;
25  import java.net.URISyntaxException;
26  
27  import com.google.gson.JsonArray;
28  import com.google.gson.JsonElement;
29  import com.google.gson.JsonParseException;
30  
31  /**
32   * Basic Json adapter that given a JsonElement builds a
33   * not null an {@link Equivalence} instance.
34   */
35  abstract class AbstractEquivalenceDeserializer {
36  
37      /**
38       * The {@code uri} constant.
39       */
40      private static final String URI = "uri";
41  
42      /**
43       * The {@code duplicates} constant.
44       */
45      private static final String DUPLICATES = "duplicates";
46  
47      /**
48       * The error message template constant.
49       */
50      private static final String EXCEPTION_MESSAGE = "URI '%s' seems to be not well-formed";
51  
52      /**
53       * Deserialize a single {@link org.nnsoft.sameas4j.Equivalence}
54       * from its Json serialization.
55       * 
56       * @param json object to be deserialized
57       * @return a not null {@link org.nnsoft.sameas4j.Equivalence} instance
58       */
59      public Equivalence getEquivalence(JsonElement json) {
60          Equivalence equivalence;
61          String uriString = json.getAsJsonObject().getAsJsonPrimitive(URI).getAsString();
62  
63          String uri = null;
64          try {
65              uri = uriString.substring(1, uriString.length() - 1);
66              equivalence = new Equivalence(new URI(uri));
67          } catch (URISyntaxException e) {
68              throw new JsonParseException(String.format(EXCEPTION_MESSAGE, uri));
69          }
70  
71          JsonArray duplicates = json.getAsJsonObject().getAsJsonArray(DUPLICATES);
72          for (int i = 0; i < duplicates.size(); i++) {
73              try {
74                  equivalence.addDuplicate(new URI(duplicates.get(i).getAsString()));
75              } catch (URISyntaxException e) {
76                  // if an equivalent URI is not well-formed it's better to do not add it, let's go on
77                  continue;
78              }
79          }
80  
81          return equivalence;
82      }
83  
84  }
