24  import java.net.URI;
25  
26  import org.nnsoft.sameas4j.cache.Cache;
27  
28  /**
29   * It models the methods that a <a href="http://sameas.org">sameas.org</a>
30   * client must expose.
31   */
32  public interface SameAsService {
33  
34      /**
35       * It returns a set of <code>URI</code> identifying equivalent
36       * resources to the input <code>URI</code>.
37       *
38       * @param uri the URI for which the equivalence has to be known.
39       * @return the URI equivalence.
40       * @throws SameAsServiceException if an HTTP or JSON parse error occurs.
41       */
42      Equivalence getDuplicates(URI uri) throws SameAsServiceException;
43  
44      /**
45       * It returns a list of {@link org.nnsoft.sameas4j.Equivalence} objects
46       * identifying equivalent <code>URI<code> for every possible co-reference
47       * match.
48       *
49       * See also the official <a href="http://sameas.org/about.php">SameAs.org documentation</a>.
50       *
51       * @param keyword the keyword for which the equivalence has to be known.
52       * @return a list of possible equivalences.
53       * @throws SameAsServiceException if an HTTP or JSON parse error occurs.
54       */
55      EquivalenceList getDuplicates(String keyword) throws SameAsServiceException;
56  
57      /**
58       * 
59       * @param cache
60       */
61      void setCache(Cache cache);
62  
63  }
