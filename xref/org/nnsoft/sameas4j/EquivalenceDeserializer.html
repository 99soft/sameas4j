31  /**
32   * Json adapter that given a JsonElement builds a
33   * not null an {@link org.nnsoft.sameas4j.Equivalence} instance.
34   */
35  final class EquivalenceDeserializer
36          extends AbstractEquivalenceDeserializer
37          implements JsonDeserializer<Equivalence> {
38  
39      /**
40       * {@inheritDoc}
41       */
42      public Equivalence deserialize(JsonElement json,
43              Type type,
44              JsonDeserializationContext context) throws JsonParseException {
45          return this.getEquivalence(json.getAsJsonArray().get(0));
46      }
