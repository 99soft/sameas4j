38  import com.google.gson.GsonBuilder;

41  /**
42   * Default implementation of {@link org.nnsoft.sameas4j.SameAsService}.
43   */
44  final class SameAsServiceImpl implements SameAsService {
45  
46      /**
47       * The sameas.org service for looking up URLs template constant.
48       */
49      private static final String SERVICE_URL = "http://sameas.org/json?uri=%s";
50  
51      /**
52       * The sameas.org service for looking up keywords template constant.
53       */
54      private static final String SERVICE_KEYWORD = "http://sameas.org/json?q=%s";
55  
56      /**
57       * The GsonBuilder used to create new equivalence JSON parser.
58       */
59      private final GsonBuilder gsonBuilder = new GsonBuilder();
60  
61      /**
62       * The {@code Cache} reference, can be null.
63       */
64      private Cache cache;
65  
66      /**
67       * Creates a new {@link org.nnsoft.sameas4j.SameAsService} instance.
68       */
69      public SameAsServiceImpl() {
70          this.gsonBuilder.registerTypeAdapter(Equivalence.class, new EquivalenceDeserializer());
71          this.gsonBuilder.registerTypeAdapter(EquivalenceList.class, new EquivalenceListDeserializer());
72      }
73  
74      /**
75       * {@inheritDoc}
76       */
77      public void setCache(Cache cache) {
78          this.cache = cache;
79      }
80  
81      /**
82       * {@inheritDoc}
83       */
84      public Equivalence getDuplicates(URI uri) throws SameAsServiceException {
85          return invokeURL(format(SERVICE_URL, uri), Equivalence.class);
86      }
87  
88      /**
89       * {@inheritDoc}
90       */
91      public EquivalenceList getDuplicates(String keyword) throws SameAsServiceException {
92          return invokeURL(format(SERVICE_KEYWORD, keyword), EquivalenceList.class);
93      }
94  
95      /**
96       * Invokes a Sameas.org service URL and parses the JSON response.
97       *
98       * @param <T> the expected return type.
99       * @param toBeInvoked the service URL has to be invoked.
100      * @param returnType the type the JSON response has to be bind to.
101      * @return the bound object.
102      * @throws SameAsServiceException is any error occurs.
103      */
104     private <T> T invokeURL(String toBeInvoked, Class<T> returnType) throws SameAsServiceException {
105         URL url;
106         try {
107             url = new URL(toBeInvoked);
108         } catch (MalformedURLException e) {
109             throw new SameAsServiceException("An error occurred while building the URL '"
110                     + toBeInvoked
111                     + "'");
112         }
113 
114         URLConnection connection = null;
115         Reader reader = null;
116         try {
117             connection = url.openConnection();
118             long lastModified = connection.getLastModified();
119 
120             if (this.cache != null) {
121                 CacheKey cacheKey = new CacheKey(toBeInvoked, lastModified);
122                 T cached = this.cache.get(cacheKey, returnType);
123                 if (cached != null) {
124                     return cached;
125                 }
126             }
127 
128             if (connection instanceof HttpURLConnection) {
129                 ((HttpURLConnection) connection).connect();
130             }
131 
132             reader = new InputStreamReader(connection.getInputStream());
133             Gson gson = this.gsonBuilder.create();
134             T response = gson.fromJson(reader, returnType);
135 
136             if (this.cache != null) {
137                 CacheKey cacheKey = new CacheKey(toBeInvoked, lastModified);
138                 this.cache.put(cacheKey, response);
139             }
140 
141             return response;
142         } catch (IOException e) {
143             throw new SameAsServiceException(format("An error occurred while invoking the URL '%s': %s",
144                     toBeInvoked,
145                     e.getMessage()));
146         } catch (JsonParseException e) {
147             throw new SameAsServiceException("An error occurred while parsing the JSON response", e);
148         } finally {
149             if (connection != null && connection instanceof HttpURLConnection) {
150                 ((HttpURLConnection) connection).disconnect();
151             }
152             if (reader != null) {
153                 try {
154                     reader.close();
155                 } catch (IOException e) {
156                     // close it quietly
157                 }
158             }
159         }
160     }
